   print('Plotting data')

   plt.xlabel('actual')
   plt.ylabel('predicted')

   plt.title('Support Vector Regression models')
   plt.legend()
   plt.show()

def predict_price(model, X_train, Y_train):
   print('Training model')
   model.fit(X_train, Y_train) # fitting the data points in the models

   print('Plotting data')
   plt.plot(Y_train, model.predict(X_train), color= 'red', label= 'X_train')
   plt.xlabel('actual')
   plt.ylabel('predicted')

   plt.title('Support Vector Regression models')
   rmse = np.sqrt(mean_squared_error(Y_train,model.predict(X_train)))
   plt.legend()
   plt.show()
   return rmse
